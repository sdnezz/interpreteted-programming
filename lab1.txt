#Метод нахождения масимального простого делителя
def MaxProstDel(num):
    i = 1
    maxdel = 1
    while i*2<=num:
        if num % i == 0:
            count = 0
            for j in range (1,i+1):
                if i%j == 0:
                    count+=1
            if count == 2 and i> maxdel:
                maxdel=i
                i+=1
            else: i+=1
        else: i+=1
    return maxdel
